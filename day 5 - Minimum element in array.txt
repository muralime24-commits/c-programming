#include<stdio.h>
int main(){
   int arr[100],n,i,minimum;
   printf("Enter number of element:");
   scanf("%d",&n);
   printf("Enter %d elements:\n",n);
   for(i=0;i<n;i++){
       scanf("%d",&arr[i]);
   }
   minimum=arr[0];
   for(i=1;i<n;i++){
       if(arr[i]>minimum)
       minimum=arr[i];
   }
   printf("Maximum element=%d\n",minimum);
   return 0;
}
