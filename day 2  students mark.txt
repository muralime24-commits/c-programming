#include <stdio.h>

// Define structure for student
struct Student {
    int roll;
    float marks;
};

int main() {
    struct Student s1, s2;
    printf("Enter roll number for Student 1: ");
    scanf("%d", &s1.roll);
    printf("Enter marks for Student 1: ");
    scanf("%f", &s1.marks);
    printf("Enter roll number for Student 2: ");
    scanf("%d", &s2.roll);
    printf("Enter marks for Student 2: ");
    scanf("%f", &s2.marks);
    printf("\n--- Student Mark List ---\n");
    printf("Roll No: %d\tMarks: %.2f\n", s1.roll, s1.marks);
    printf("Roll No: %d\tMarks: %.2f\n", s2.roll, s2.marks);

    return 0;
}
