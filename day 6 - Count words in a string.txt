#include <stdio.h>
int main() {
    char str[200];
    int i = 0, word_count = 0;
    printf("Enter a string: ");
    fgets(str, sizeof(str), stdin);
    while (str[i] != '\0') {
        if ((str[i] != ' ' && str[i] != '\n') &&
            (i == 0 || str[i - 1] == ' ' || str[i - 1] == '\n')) {
            word_count++;
        }
        i++;
    }
    printf("Total number of words: %d\n", word_count);
    return 0;
}
