#include <stdio.h>
int main() {
    int A[2][2] = {{1, 2}, {3, 4}};
    int B[2][2] = {{5, 6}, {7, 8}};
    int C[2][2] = {{9, 10}, {11, 12}};
    int AB[2][2], ABC[2][2];
    for (int i = 0; i < 2; i++)
        for (int j = 0; j < 2; j++) {
            AB[i][j] = 0;
            for (int k = 0; k < 2; k++)
                AB[i][j] += A[i][k] * B[k][j];
        }
    for (int i = 0; i < 2; i++)
        for (int j = 0; j < 2; j++) {
            ABC[i][j] = 0;
            for (int k = 0; k < 2; k++)
                ABC[i][j] += AB[i][k] * C[k][j];
        }
    for (int i = 0; i < 2; i++) {
        for (int j = 0; j < 2; j++)
            printf("%d ", ABC[i][j]);
        printf("\n");
    }
    return 0;
}
