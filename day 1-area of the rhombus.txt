#include <stdio.h>
int main() {
    float d1, d2, area;
    printf("Enter the length of first diagonal (d1): ");
    scanf("%f", &d1);
    printf("Enter the length of second diagonal (d2): ");
    scanf("%f", &d2);
    area = (d1 * d2) / 2;
    printf("Area of the rhombus = %.2f\n", area);
    return 0;
}
