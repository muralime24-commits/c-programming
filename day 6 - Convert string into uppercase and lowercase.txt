#include <stdio.h>
int main() {
    char str[100];
    int i = 0;
    printf("Enter a string: ");
    fgets(str, sizeof(str), stdin);
    printf("Uppercase: ");
    while (str[i] != '\0') {
        if (str[i] >= 'a' && str[i] <= 'z') {
            printf("%c", str[i] - 32);
        } else {
            printf("%c", str[i]);
        }
        i++;
    }
    i = 0;
    printf("\nLowercase: ");
    while (str[i] != '\0') {
        if (str[i] >= 'A' && str[i] <= 'Z') {
            printf("%c", str[i] + 32);
        } else {
            printf("%c", str[i]);
        }
        i++;
    }

    return 0;
}
