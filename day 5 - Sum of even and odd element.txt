#include <stdio.h>
int main() {
    int arr[] = {3, 6, 9, 2, 5, 8};
    int size = sizeof(arr) / sizeof(arr[0]);
    int evenSum = 0, oddSum = 0;
    for (int i = 0; i < size; i++) {
        if (arr[i] % 2 == 0)
            evenSum += arr[i];
        else
            oddSum += arr[i];
    }
    printf("Sum of even elements: %d\n", evenSum);
    printf("Sum of odd elements: %d\n", oddSum);
    return 0;
}
